Horner's Method

计算多项式的值

function Horner(c,a){
    //Calc y=A(n)X^n+A(n-1)X^(n-1)+...+A1X+A0 when x=c
    //a is the array[a0,a1,a2,...,an]
    let n=a.length;//n+1
    let y=a[n-1];
    for(let i=1;i<n;i++){
        y=y*c+a[n-1-i];
    }
    return y;
}
